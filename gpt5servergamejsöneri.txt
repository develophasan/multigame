2.7 Sunucu tick: mermi ilerletme, çarpışma, hasar, ölüm/respawn, maç süresi
+// ---- Sunucu Tick Döngüsü (60 Hz) ----
+setInterval(() => {
+  const now = Date.now();
+  for (const [gameId, game] of activeGames) {
+    if (game.gameState !== 'playing') continue;
+
+    // maç süresi
+    if (game.startedAt) {
+      const elapsed = Math.floor((now - game.startedAt) / 1000);
+      game.remaining = Math.max(0, MATCH_SECONDS - elapsed);
+      if (game.remaining <= 0) {
+        game.gameState = 'lobby';
+        for (const p of game.players) p.isReady = false;
+        io.to(gameId).emit('return-to-lobby', { message: 'Süre bitti, lobiye dönüldü.' });
+        continue;
+      }
+    }
+
+    // mermiler
+    const dt = 1 / TICK_HZ;
+    const keep = [];
+    for (const pr of game.projectiles) {
+      pr.x += pr.vx * dt;
+      pr.y += pr.vy * dt;
+      // sınır
+      if (pr.x<0||pr.x>CANVAS.w||pr.y<0||pr.y>CANVAS.h) continue;
+
+      const target = game.players.find(p => p.socketId === pr.targetId);
+      if (!target) continue;
+      if (target.deadUntil && now < target.deadUntil) { keep.push(pr); continue; }
+
+      // çarpışma
+      const hit = Math.hypot(pr.x - target.position.x, pr.y - target.position.y) <= (PLAYER_RADIUS + PROJECTILE_RADIUS);
+      if (hit) {
+        const owner = game.players.find(p => p.socketId === pr.ownerId);
+        const skill = SKILLS[pr.skillKey] || SKILLS.fire;
+        if (owner && target) {
+          target.health = (typeof target.health==='number' ? target.health : 100) - skill.dmg;
+          owner.stats = owner.stats || { damage:0,kills:0,deaths:0 };
+          owner.stats.damage += skill.dmg;
+          io.to(gameId).emit('player-took-damage', { targetId: target.socketId, newHealth: Math.max(0, target.health) });
+
+          if (target.health <= 0) {
+            target.stats = target.stats || { damage:0,kills:0,deaths:0 };
+            target.stats.deaths += 1;
+            owner.stats.kills += 1;
+            target.deadUntil = now + RESPAWN_MS;
+            io.to(gameId).emit('player-died', { targetId: target.socketId, killerId: owner.socketId });
+          }
+        }
+        continue; // mermi yok oldu
+      }
+      keep.push(pr);
+    }
+    game.projectiles = keep;
+
+    // respawn
+    for (const p of game.players) {
+      if (p.deadUntil && now >= p.deadUntil) {
+        p.deadUntil = null;
+        p.health = 100;
+        p.position = randSpawn();
+        io.to(gameId).emit('player-moved', { playerId: p.socketId, position: p.position });
+      }
+    }
+  }
+}, 1000 / TICK_HZ);

Benim verdiğim game.js sürümü zaten senin event’lerinle uyumlu:

player-move (sende var)

player-shoot (sende var, artık sunucu otoriter yaptı)

player-took-damage, player-died, projectile-fired (sende var)

game-started, player-joined, player-left, player-ready-changed, player-shape-changed (var)

Ek olarak game.js’te player-damage gönderilmiyor; bu endpoint’i kapattık (sunucu hesaplıyor).

change-shape’daki .save() hatasını gider.

player-damage’ı kaldır; hasarı server tick içinde mermi-çarpışmasıyla hesapla.

Oyun başladığında HP/istatistik reseti ve rastgele spawn.
